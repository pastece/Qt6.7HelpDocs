             id: pointLight
             position: Qt.vector3d(200, 200, 400)
             brightness: 50
             ambientColor: Qt.rgba(0.5, 0.3, 0.1, 1.0)
             SequentialAnimation {
                 loops: Animation.Infinite
                 running: true
                 NumberAnimation {
                     target: pointLight
                     property: "brightness"
                     to: 400
                     duration: 2000
                     easing.type: Easing.OutElastic

             source: "#Cube"
             position: Qt.vector3d(-250, 150, 100)
